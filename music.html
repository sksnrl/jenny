let songPool = []; let playedSongIds = new Set(); let currentSong = null; let stage = 1; let isFinished = false; const audioPlayer = document.getElementById('music-player'); function syncVolume(val) { audioPlayer.volume = val; document.getElementById('lobby-volume').value = val; document.getElementById('game-volume').value = val; } // 현재 연도(2026년)를 자동으로 계산하여 최신곡 유실 방지 const CURRENT_YEAR = new Date().getFullYear(); async function initGameFlow(isNewSetting) { const loading = document.getElementById('loading'); const selectedDecades = Array.from(document.querySelectorAll('.year-check:checked')).map(cb => parseInt(cb.value)); const genre = document.getElementById('genre-select').value; if (selectedDecades.length === 0) { alert("최소 하나 이상의 시대를 선택해주세요!"); return; } loading.style.display = 'flex'; if (isNewSetting) { document.getElementById('lobby').style.display = 'none'; document.getElementById('game').style.display = 'block'; playedSongIds.clear(); songPool = []; } resetGameState(); if (songPool.length === 0) { try { let allYearsPromises = []; // 선택한 각 10년 단위(Decade)에 대해 내부 연도별로 TOP 100씩 루프 추출 selectedDecades.forEach(decade => { // 해당 시대의 시작 연도부터 끝 연도 계산 (예: 2020년대 -> 2020 ~ 2026) const startYear = decade; const endYear = Math.min(decade + 9, CURRENT_YEAR); for (let year = startYear; year <= endYear; year++) { let term = ''; // 장르 및 연도별 최적의 인기 키워드 검색 조합 if (genre === 'kpop') { term = `${year} K-Pop`; } else if (genre === 'pop') { term = `${year} Pop`; } else { term = `${year} ${genre}`; } // 각 연도별로 가장 인기 있는 Top 100 곡을 확보하기 위해 limit=100 설정 const url = `https://itunes.apple.com/search?term=${encodeURIComponent(term)}&country=kr&entity=song&limit=100&sort=recent`; const fetchPromise = fetch(url) .then(r => r.json()) .then(data => { // 엄격한 연도 필터링: 해당 연도에 정확히 발매된 곡만 필터링 return data.results.filter(song => { if (!song.releaseDate) return false; return new Date(song.releaseDate).getFullYear() === year; }); }) .catch(err => { console.error(`${year}년 데이터 호출 오류:`, err); return []; }); allYearsPromises.push(fetchPromise); } }); document.getElementById('load-text').innerText = "연도별 TOP 100 곡 수집 중..."; const resultsPerYear = await Promise.all(allYearsPromises); let tempPool = []; resultsPerYear.forEach(songs => { tempPool.push(...songs); }); // 전체 수집된 데이터에서 중복 트랙 제거 및 무작위 셔플 songPool = Array.from(new Map(tempPool.map(s => [s.trackId, s])).values()).sort(() => Math.random() - 0.5); if (songPool.length === 0) { alert("선택하신 조건의 연도별 TOP 100 데이터를 충분히 찾을 수 없습니다. 장르를 바꾸거나 다시 시도해주세요."); goBackToLobby(); return; } } catch (e) { console.error(e); alert("데이터를 빌드하는 과정에서 오류가 발생했습니다."); goBackToLobby(); return; } } // 풀(Pool)에서 이미 플레이하지 않은 곡 선택 currentSong = songPool.find(s => !playedSongIds.has(s.trackId)) || songPool[0]; playedSongIds.add(currentSong.trackId); prepareGameContent(); loading.style.display = 'none'; } function prepareGameContent() { const year = new Date(currentSong.releaseDate).getFullYear(); document.getElementById('display-info').innerText = currentSong.primaryGenreName; document.getElementById('hint-1').innerText = `${currentSong.artistName} (${year}년 발매)`; document.getElementById('hint-1').classList.add('visible'); document.getElementById('hint-2').innerText = currentSong.trackName.split('').map(c => { if (Hangul.isHangul(c)) return Hangul.disassemble(c)[0]; if (/[a-zA-Z\u3040-\u30ff\u3400-\u4dbf\u4e00-\u9fff]/.test(c)) return Math.random() < 0.3 ? c : '_'; return c; }).join(''); audioPlayer.src = currentSong.previewUrl; } function nextHint() { if (stage >= 4) return; stage++; document.querySelectorAll('.stage-dot')[stage-1].classList.add('active'); if (stage === 2) document.getElementById('hint-2').classList.add('visible'); else if (stage === 3) document.getElementById('btn-2s').classList.add('active'); else if (stage === 4) { document.getElementById('btn-3s').classList.add('active'); document.getElementById('next-hint-btn').style.display = 'none'; } } function playAudio(ms) { audioPlayer.currentTime = 0; audioPlayer.play(); setTimeout(() => { if(!isFinished) audioPlayer.pause(); }, ms); } function checkAnswer() { if (isFinished) return; isFinished = true; const userVal = document.getElementById('user-answer').value.replace(/\s/g, '').toLowerCase(); const correctVal = currentSong.trackName.replace(/\s/g, '').toLowerCase(); document.getElementById('user-answer').disabled = true; document.getElementById('submit-btn').disabled = true; document.getElementById('next-hint-btn').style.display = 'none'; const isCorrect = userVal === correctVal; document.getElementById('result-msg').innerHTML = isCorrect ? `정답! 🎉` : `오답 😢 (정답: ${currentSong.trackName})`; document.getElementById('hint-2').innerText = currentSong.trackName; document.getElementById('hint-2').classList.add('visible'); document.getElementById('hint-2').style.color = "var(--success)"; audioPlayer.currentTime = 0; audioPlayer.play(); document.getElementById('next-song-btn').style.display = 'block'; } function resetGameState() { stage = 1; isFinished = false; document.getElementById('result-msg').innerHTML = ""; document.getElementById('user-answer').value = ""; document.getElementById('user-answer').disabled = false; document.getElementById('submit-btn').disabled = false; document.getElementById('next-song-btn').style.display = 'none'; document.getElementById('next-hint-btn').style.display = 'block'; document.querySelectorAll('.btn-audio').forEach((b, i) => i === 0 ? b.classList.add('active') : b.classList.remove('active')); document.querySelectorAll('.hint-text').forEach(t => t.classList.remove('visible')); document.querySelectorAll('.stage-dot').forEach((d, i) => i === 0 ? d.classList.add('active') : d.classList.remove('active')); } function goBackToLobby() { audioPlayer.pause(); document.getElementById('game').style.display = 'none'; document.getElementById('lobby').style.display = 'block'; }
